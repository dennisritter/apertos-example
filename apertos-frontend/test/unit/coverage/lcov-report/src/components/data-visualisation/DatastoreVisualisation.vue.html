<template>
  <div class="datastore-visualisation">
    <svg width="100%" height="500px" ref="svg">
      <g>
        <path class="line" :d="graph.paths.line"/>
        <!--<path class="selector" :d="graph.paths.selector"/>-->
      </g>
      <g class="axis-x" ref="xaxis" :style="translateAxis({ x: 0, y: graph.height - margin.bottom })"></g>
      <g class="axis-y" ref="yaxis" :style="translateAxis({ x: legend.width + margin.left, y: 0 })"></g>
    </svg>
  </div>
</template>
 
<script>
  /* eslint-disable max-len */
  import { mapActions, mapGetters } from 'vuex';
  import * as d3 from 'd3';
 
  export default {
    name: 'datastoreVisualisation',
    dependencies: 'DatastoreService',
    props: {
      distId: {
        type: String,
        required: true,
      },
    },
    data() {
      return {
        height: 0,
        width: 0,
        margin: {
          top: 10,
          right: 10,
          bottom: 0,
          left: 0,
        },
        data: {
          dates: [],
          values: [],
        },
        graph: {
          height: 0,
          width: 0,
          line: '',
          paths: {
            area: '',
            line: '',
            selector: '',
          },
          points: [],
          scaled: {
            x: null,
            y: null,
          },
        },
        legend: {
          height: 80,
          width: 50,
          paths: {
            line: '',
          },
          points: [],
        },
      };
    },
    computed: {
      ...mapGetters('datastoreVisualisation', [
        'getId',
        'getRecords',
      ]),
      getData() {
        const records = this.getRecords;
        const dates = [];
        const values = [];
        for (const record of records) {
          dates.push(new Date(record.date));
          values.push(record.value);
        }
        // Sort dates before using them.
        // Sometimes Dates do not appear in the right order which leads to weird behaviour in the chart
        dates.sort((a, b) => b - a);
        return { dates, values };
      },
//      padded() {
//        const width = this.width - this.margin.left - this.margin.right;
//        const height = this.height - this.margin.top - this.margin.bottom;
//        return { width, height };
//      },
    },
    methods: {
      ...mapActions('datastoreVisualisation', [
        'loadData',
        'useService',
      ]),
      translateAxis({ x, y }) {
        return { transform: `translate(${x}px, ${y}px)` };
      },
      onResize() {
        this.calcDimensions();
      },
      calcDimensions() {
        const boundingClientRect = this.$refs.svg.getBoundingClientRect();
        this.width = boundingClientRect.width;
        this.height = boundingClientRect.height;
 
 
        this.graph.width = this.width - this.legend.width - this.margin.right - this.margin.left;
        this.graph.height = this.height - this.legend.height - this.margin.top - this.margin.bottom;
      },
      // createArea: d3.area().x(d => d.x).y0(d => d.max).y1(d => d.y),
      createLine: d3.line().x(d => d.x).y(d => d.y),
      createValueSelector: d3.area().x(d => d.x).y0(d => d.max).y1(0),
      init() {
        this.calcScales();
        this.drawGraph();
      },
      update() {
        this.calcScales();
        this.drawGraph();
      },
      calcScales() {
        this.graph.scaled.x = d3.scaleLinear().range([this.legend.width + this.margin.left, this.width - this.margin.right]);
        this.graph.scaled.y = d3.scaleLinear().range([this.graph.height - this.margin.bottom, this.margin.top]);
        d3.axisLeft().scale(this.graph.scaled.y);
        d3.axisBottom().scale(this.graph.scaled.x);
      },
      drawGraph() {
        this.graph.scaled.x.domain(d3.extent(this.data.dates, d => d));
        this.graph.scaled.y.domain(d3.extent(this.data.values, d => d));
        this.graph.points = [];
        for (let i = 0; i < this.data.values.length; i += 1) {
          this.graph.points.push({
            x: this.graph.scaled.x(this.data.dates[i]),
            y: this.graph.scaled.y(this.data.values[i]),
            max: this.graph.height,
          });
        }
        this.graph.paths.line = this.createLine(this.graph.points);
        const xAxis = d3.axisBottom(this.graph.scaled.x).ticks(30).tickFormat(d3.timeFormat('%X'));
        const yAxis = d3.axisLeft(this.graph.scaled.y).tickFormat(d => d);
        const $xaxis = d3.select(this.$refs.xaxis);
        const $yaxis = d3.select(this.$refs.yaxis);
        $xaxis.call(xAxis)
        .selectAll('text')
        .style('text-anchor', 'end')
        .attr('dx', '-.8em')
        .attr('dy', '.15em')
        .attr('transform', 'rotate(-65)');
        $yaxis.call(yAxis);
      },
    },
    created() {
      // TODO: Replace asap to detect changed from backend automatically
//      setInterval(() => {
//        this.loadData(this.getId);
//      }, 500);
    },
    mounted() {
      this.useService(this.DatastoreService);
      if (this.distId) {
        this.loadData(this.distId).then(() => {
          this.calcDimensions();
        });
      } else if (this.$route.params.dist_id) {
        this.loadData(this.$route.params.dist_id).then(() => {
          this.calcDimensions();
        });
      }
      window.addEventListener('resize', this.onResize);
//      this.init();
    },
    updated() {
      if (this.distId) {
        this.loadData(this.distId).then(() => {
          this.calcDimensions();
        });
      } else if (this.$route.params.dist_id) {
        this.loadData(this.$route.params.dist_id).then(() => {
          this.calcDimensions();
        });
      }
    },
    beforeDestroy() {
      window.removeEventListener('resize', this.onResize);
    },
    watch: {
      getRecords: function dataChanged() {
        this.data = this.getData;
        this.update();
      },
      width: function widthChanged() {
        this.update();
      },
    },
  };
</script>
